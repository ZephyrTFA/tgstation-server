243 public async ValueTask<ISessionController> LaunchNew(
244 IDmbProvider dmbProvider,
245 IByondExecutableLock currentByondLock,
246 DreamDaemonLaunchParameters launchParameters,
247 bool apiValidate,
248 CancellationToken cancellationToken)
249 {
250 logger.LogTrace("Begin session launch...");
251 if (!launchParameters.Port.HasValue)
252 throw new InvalidOperationException("Given port is null!");
253 switch (dmbProvider.CompileJob.MinimumSecurityLevel)
254 {
255 case DreamDaemonSecurity.Ultrasafe:
256 break;
257 case DreamDaemonSecurity.Safe:
258 if (launchParameters.SecurityLevel == DreamDaemonSecurity.Ultrasafe)
259 {
260 logger.LogTrace("Boosting security level to minimum of Safe");
261 launchParameters.SecurityLevel = DreamDaemonSecurity.Safe;
262 }
263
264 break;
265 case DreamDaemonSecurity.Trusted:
266 if (launchParameters.SecurityLevel != DreamDaemonSecurity.Trusted)
267 logger.LogTrace("Boosting security level to minimum of Trusted");
268
269 launchParameters.SecurityLevel = DreamDaemonSecurity.Trusted;
270 break;
271 default:
272 throw new InvalidOperationException(String.Format(CultureInfo.InvariantCulture, "Invalid DreamDaemonSecurity value: {0}", dmbProvider.CompileJob.MinimumSecurityLevel));
273 }
274
275 // get the byond lock
276 var byondLock = currentByondLock ?? await byond.UseExecutables(
277 Version.Parse(dmbProvider.CompileJob.ByondVersion),
278 gameIOManager.ConcatPath(dmbProvider.Directory, dmbProvider.DmbName),
279 cancellationToken);
280 try
281 {
282 logger.LogDebug(
283 "Launching session with CompileJob {compileJobId}...",
284 dmbProvider.CompileJob.Id);
285
286 PortBindTest(launchParameters.Port.Value);
287 await CheckPagerIsNotRunning();
288
289 string outputFilePath = null;
290 var preserveLogFile = true;
291
292 var cliSupported = byondLock.SupportsCli;
293 if (launchParameters.LogOutput.Value)
294 {
295 var now = DateTimeOffset.UtcNow;
296 var dateDirectory = diagnosticsIOManager.ConcatPath(DreamDaemonLogsPath, now.ToString("yyyy-MM-dd", CultureInfo.InvariantCulture));
297 await diagnosticsIOManager.CreateDirectory(dateDirectory, cancellationToken);
298 outputFilePath = diagnosticsIOManager.ResolvePath(
299 diagnosticsIOManager.ConcatPath(
300 dateDirectory,
301 $"dd-utc-{now.ToString("yyyy-MM-dd-HH-mm-ss", CultureInfo.InvariantCulture)}{(apiValidate ? "-dmapi" : String.Empty)}.log"));
302
303 logger.LogInformation("Logging DreamDaemon output to {path}...", outputFilePath);
304 }
305 else if (!cliSupported)
306 {
307 outputFilePath = gameIOManager.ConcatPath(dmbProvider.Directory, $"{Guid.NewGuid()}.dd.log");
308 preserveLogFile = false;
309 }
310
311 var accessIdentifier = cryptographySuite.GetSecureString();
312
313 var byondTopicSender = topicClientFactory.CreateTopicClient(
314 TimeSpan.FromMilliseconds(
315 launchParameters.TopicRequestTimeout.Value));
316
317 if (!apiValidate && dmbProvider.CompileJob.DMApiVersion == null)
318 logger.LogDebug("Session will have no DMAPI support!");
319
320 // launch dd
321 var process = await CreateDreamDaemonProcess(
322 dmbProvider,
323 byondTopicSender,
324 byondLock,
325 launchParameters,
326 accessIdentifier,
327 outputFilePath,
328 apiValidate,
329 cancellationToken);
330
331 try
332 {
333 var chatTrackingContext = chat.CreateTrackingContext();
334
335 try
336 {
337 var runtimeInformation = CreateRuntimeInformation(
338 dmbProvider,
339 chatTrackingContext,
340 launchParameters.SecurityLevel.Value,
341 launchParameters.Visibility.Value,
342 apiValidate);
343
344 var reattachInformation = new ReattachInformation(
345 dmbProvider,
346 process,
347 runtimeInformation,
348 accessIdentifier,
349 launchParameters.Port.Value);
350
351 var sessionController = new SessionController(
352 reattachInformation,
353 instance,
354 process,
355 byondLock,
356 byondTopicSender,
357 chatTrackingContext,
358 bridgeRegistrar,
359 chat,
360 assemblyInformationProvider,
361 asyncDelayer,
362 loggerFactory.CreateLogger<SessionController>(),
363 () => LogDDOutput(
364 process,
365 outputFilePath,
366 cliSupported,
367 preserveLogFile,
368 CancellationToken.None), // DCT: None available
369 launchParameters.StartupTimeout,
370 false,
371 apiValidate);
372
373 return sessionController;
374 }
375 catch
376 {
377 chatTrackingContext.Dispose();
378 throw;
379 }
380 }
381 catch
382 {
383 await using (process)
384 {
385 process.Terminate();
386 await process.Lifetime;
387 throw;
388 }
389 }
390 }
391 catch
392 {
393 if (currentByondLock == null)
394 byondLock.Dispose();
395 throw;
396 }
397 }

402 ReattachInformation reattachInformation,
403 CancellationToken cancellationToken)
404 {
405 ArgumentNullException.ThrowIfNull(reattachInformation);
406
407 logger.LogTrace("Begin session reattach...");
408 var byondTopicSender = topicClientFactory.CreateTopicClient(reattachInformation.TopicRequestTimeout);
409 var byondLock = await byond.UseExecutables(
410 Version.Parse(reattachInformation.Dmb.CompileJob.ByondVersion),
411 null, // Doesn't matter if it's trusted or not on reattach
412 cancellationToken);
413
414 try
415 {
416 logger.LogDebug(
417 "Attaching to session PID: {pid}, CompileJob: {compileJobId}...",
418 reattachInformation.ProcessId,
419 reattachInformation.Dmb.CompileJob.Id);
420
421 var process = processExecutor.GetProcess(reattachInformation.ProcessId);
422 if (process == null)
423 return null;
424
425 try
426 {
427 if (!byondLock.SupportsCli)
428 networkPromptReaper.RegisterProcess(process);
429
430 var chatTrackingContext = chat.CreateTrackingContext();
431 try
432 {
433 var runtimeInformation = CreateRuntimeInformation(
434 reattachInformation.Dmb,
435 chatTrackingContext,
436 reattachInformation.LaunchSecurityLevel,
437 reattachInformation.LaunchVisibility,
438 false);
439 reattachInformation.SetRuntimeInformation(runtimeInformation);
440
441 var controller = new SessionController(
442 reattachInformation,

448 bridgeRegistrar,

450 assemblyInformationProvider,
451 asyncDelayer,
452 loggerFactory.CreateLogger<SessionController>(),
453 () => ValueTask.CompletedTask,
454 null,
455 true,
456 false);
457
458 process = null;
459 byondLock = null;
460 chatTrackingContext = null;
461
462 return controller;
463 }
464 catch

466 chatTrackingContext.Dispose();
467 throw;
468 }
469 }

478 byondLock.Dispose();
479 throw;

495 async ValueTask<IProcess> CreateDreamDaemonProcess(
496 IDmbProvider dmbProvider,
497 ITopicClient byondTopicSender,
498 IByondExecutableLock byondLock,
499 DreamDaemonLaunchParameters launchParameters,
500 string accessIdentifier,
501 string logFilePath,
502 bool apiValidate,
503 CancellationToken cancellationToken)
504 {
505 // set command line options
506 // more sanitization here cause it uses the same scheme
507 var parameters = $"{DMApiConstants.ParamApiVersion}={byondTopicSender.SanitizeString(DMApiConstants.InteropVersion.Semver().ToString())}&{byondTopicSender.SanitizeString(DMApiConstants.ParamServerPort)}={serverPortProvider.HttpApiPort}&{byondTopicSender.SanitizeString(DMApiConstants.ParamAccessIdentifier)}={byondTopicSender.SanitizeString(accessIdentifier)}";
508
509 if (!String.IsNullOrEmpty(launchParameters.AdditionalParameters))
510 parameters = $"{parameters}&{launchParameters.AdditionalParameters}";
511
512 // important to run on all ports to allow port changing
513 var arguments = String.Format(
514 CultureInfo.InvariantCulture,
515 "{0} -port {1} -ports 1-65535 {2}-close -verbose -{3} -{4}{5}{6}{7} -params \"{8}\"",
516 dmbProvider.DmbName,
517 launchParameters.Port.Value,
518 launchParameters.AllowWebClient.Value ? "-webclient " : String.Empty,
519 SecurityWord(launchParameters.SecurityLevel.Value),
520 VisibilityWord(launchParameters.Visibility.Value),
521 !byondLock.SupportsCli
522 ? $" -logself -log {logFilePath}"
523 : String.Empty, // DD doesn't output anything if -logself is set???
524 launchParameters.StartProfiler.Value
525 ? " -profile"
526 : String.Empty,
527 byondLock.SupportsMapThreads && launchParameters.MapThreads.Value != 0
528 ? $" -map-threads {launchParameters.MapThreads.Value}"
529 : String.Empty,
530 parameters);
531
532 var process = processExecutor.LaunchProcess(
533 byondLock.DreamDaemonPath,
534 dmbProvider.Directory,
535 arguments,
536 logFilePath,
537 byondLock.SupportsCli,
538 true);
539
540 try
541 {
542 if (!apiValidate)
543 {
544 if (sessionConfiguration.HighPriorityLiveDreamDaemon)
545 process.AdjustPriority(true);
546 }
547 else if (sessionConfiguration.LowPriorityDeploymentProcesses)
548 process.AdjustPriority(false);
549
550 if (!byondLock.SupportsCli)
551 networkPromptReaper.RegisterProcess(process);
552
553 // If this isnt a staging DD (From a Deployment), fire off an event
554 if (!apiValidate)
555 await eventConsumer.HandleEvent(
556 EventType.DreamDaemonLaunch,
557 new List<string>
558 {
559 process.Id.ToString(CultureInfo.InvariantCulture),
560 },
561 false,
562 cancellationToken);
563
564 return process;
565 }
566 catch
567 {
568 await using (process)
569 {
570 process.Terminate();
571 await process.Lifetime;
572 throw;
573 }
574 }
575 }

586 async ValueTask LogDDOutput(IProcess process, string outputFilePath, bool cliSupported, bool preserveFile, CancellationToken cancellationToken)
587 {
588 try
589 {
590 string ddOutput = null;
591 if (cliSupported)
592 ddOutput = await process.GetCombinedOutput(cancellationToken);
593
594 if (ddOutput == null)
595 try
596 {
597 var dreamDaemonLogBytes = await gameIOManager.ReadAllBytes(
598 outputFilePath,
599 cancellationToken);
600
601 ddOutput = Encoding.UTF8.GetString(dreamDaemonLogBytes);
602 }
603 finally
604 {
605 if (!preserveFile)
606 try
607 {
608 logger.LogTrace("Deleting temporary log file {path}...", outputFilePath);
609 await gameIOManager.DeleteFile(outputFilePath, cancellationToken);
610 }
611 catch (Exception ex)
612 {
613 logger.LogWarning(ex, "Failed to delete DreamDaemon log file {outputFilePath}!", outputFilePath);
614 }
615 }
616
617 logger.LogTrace(
618 "DreamDaemon Output:{newLine}{output}",
619 Environment.NewLine,
620 ddOutput);
621 }
622 catch (Exception ex)
623 {
624 logger.LogWarning(ex, "Error reading DreamDaemon output!");
625 }

657 async ValueTask CheckPagerIsNotRunning()
658 {
659 if (!platformIdentifier.IsWindows)
660 return;
661
662 await using var otherProcess = processExecutor.GetProcessByName("byond");
663 if (otherProcess == null)
664 return;
665
666 var otherUsername = otherProcess.GetExecutingUsername();
667
668 await using var ourProcess = processExecutor.GetCurrentProcess();
669 var ourUsername = ourProcess.GetExecutingUsername();
670
671 if (otherUsername.Equals(ourUsername, StringComparison.Ordinal))
672 throw new JobException(ErrorCode.DeploymentPagerRunning);
673 }
