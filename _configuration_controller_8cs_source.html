357 bool ForbidDueToModeConflicts(string path, out ISystemIdentity systemIdentityToUse)
358 {
359 if (Instance.ConfigurationType == ConfigurationType.Disallowed
360 || (Instance.ConfigurationType == ConfigurationType.SystemIdentityWrite && AuthenticationContext.SystemIdentity == null)
361 || (path != null && ioManager.PathContainsParentAccess(path)))
362 {
363 systemIdentityToUse = null;
364 return true;
