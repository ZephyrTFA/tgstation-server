104 public override async Task OnActionExecutionAsync(ActionExecutingContext context, ActionExecutionDelegate next)
105 {
106 ArgumentNullException.ThrowIfNull(context);
107
108 // ALL valid token and login requests that match a route go through this function
109 // 404 is returned before
110 if (AuthenticationContext != null && AuthenticationContext.User == null)
111 {
112 // valid token, expired password
113 await Unauthorized().ExecuteResultAsync(context);
114 return;
115 }
116
117 // validate the headers
118 try
119 {
120 ApiHeaders = new ApiHeaders(Request.GetTypedHeaders());
121
122 if (!ApiHeaders.Compatible())
123 {
124 await this.StatusCode(
125 HttpStatusCode.UpgradeRequired,
126 new ErrorMessageResponse(ErrorCode.ApiMismatch))
127 .ExecuteResultAsync(context);
128 return;
129 }
130
131 var errorCase = await ValidateRequest(context.HttpContext.RequestAborted);
132 if (errorCase != null)
133 {
134 await errorCase.ExecuteResultAsync(context);
135 return;
136 }
137 }
138 catch (HeadersException)
139 {
140 if (requireHeaders)
141 {
142 await HeadersIssue(false)
143 .ExecuteResultAsync(context);
144 return;
145 }
146 }
147
148 if (ModelState?.IsValid == false)
149 {
150 var errorMessages = ModelState
151 .SelectMany(x => x.Value.Errors)
152 .Select(x => x.ErrorMessage)
153
154 // We use RequiredAttributes purely for preventing properties from becoming nullable in the databases
155 // We validate missing required fields in controllers
156 // Unfortunately, we can't remove the whole validator for that as it checks other things like StringLength
157 // This is the best way to deal with it unfortunately
158 .Where(x => !x.EndsWith(" field is required.", StringComparison.Ordinal));
159
160 if (errorMessages.Any())
161 {
162 await BadRequest(
163 new ErrorMessageResponse(ErrorCode.ModelValidationFailure)
164 {
165 AdditionalData = String.Join(Environment.NewLine, errorMessages),
166 })
167 .ExecuteResultAsync(context);
168 return;
169 }
170
171 ModelState.Clear();
172 }
173
174 using (ApiHeaders?.InstanceId != null
175 ? LogContext.PushProperty(SerilogContextHelper.InstanceIdContextProperty, ApiHeaders.InstanceId)
176 : null)
177 using (AuthenticationContext != null
178 ? LogContext.PushProperty(SerilogContextHelper.UserIdContextProperty, AuthenticationContext.User.Id)
179 : null)
180 using (LogContext.PushProperty(SerilogContextHelper.RequestPathContextProperty, $"{Request.Method} {Request.Path}"))
181 {
182 if (ApiHeaders != null)
183 {
184 var isGet = HttpMethods.IsGet(Request.Method);
185 if (!(isGet && Request.Path.StartsWithSegments(Routes.Jobs, StringComparison.OrdinalIgnoreCase)))
186 Logger.Log(
187 isGet
188 ? LogLevel.Trace
189 : LogLevel.Debug,
190 "Starting API request: Version: {clientApiVersion}. {userAgentHeaderName}: {clientUserAgent}",
191 ApiHeaders.ApiVersion.Semver(),
192 HeaderNames.UserAgent,
193 ApiHeaders.RawUserAgent);
194 }
195 else if (Request.Headers.TryGetValue(HeaderNames.UserAgent, out var userAgents))
196 Logger.LogDebug(
197 "Starting unauthorized API request. {userAgentHeaderName}: {allUserAgents}",
198 HeaderNames.UserAgent,
199 userAgents);
200 else
201 Logger.LogDebug(
202 "Starting unauthorized API request. No {userAgentHeaderName}!",
203 HeaderNames.UserAgent);
204 await base.OnActionExecutionAsync(context, next);
205 }
206 }

275 throw new InvalidOperationException("Expected a header parse exception!");

351 async ValueTask<IActionResult> PaginatedImpl<TModel, TResultModel>(
352 Func<ValueTask<PaginatableResult<TModel>>> queryGenerator,
353 Func<TResultModel, ValueTask> resultTransformer,
354 int? pageQuery,
355 int? pageSizeQuery,
356 CancellationToken cancellationToken)
357 {
358 ArgumentNullException.ThrowIfNull(queryGenerator);
359
360 if (pageQuery <= 0 || pageSizeQuery <= 0)
361 return BadRequest(new ErrorMessageResponse(ErrorCode.ApiInvalidPageOrPageSize));
362
363 var pageSize = pageSizeQuery ?? DefaultPageSize;
364 if (pageSize > MaximumPageSize)
365 return BadRequest(new ErrorMessageResponse(ErrorCode.ApiPageTooLarge)
366 {
367 AdditionalData = $"Maximum page size: {MaximumPageSize}",
368 });
369
370 var page = pageQuery ?? 1;
371
372 var paginationResult = await queryGenerator();
373 if (paginationResult.EarlyOut != null)
374 return paginationResult.EarlyOut;
375
376 var queriedResults = paginationResult
377 .Results
378 .Skip((page - 1) * pageSize)
379 .Take(pageSize);
380
381 int totalResults;
382 List<TModel> pagedResults;
383 if (queriedResults.Provider is IAsyncQueryProvider)
384 {
385 totalResults = await paginationResult.Results.CountAsync(cancellationToken);
386 pagedResults = await queriedResults
387 .ToListAsync(cancellationToken);
388 }
389 else
390 {
391 totalResults = paginationResult.Results.Count();
392 pagedResults = queriedResults.ToList();
393 }
394
395 ICollection<TResultModel> finalResults;
396 if (typeof(TModel) == typeof(TResultModel))
397 finalResults = (List<TResultModel>)(object)pagedResults; // clearly a safe cast
398 else
399 finalResults = pagedResults
400 .OfType<IApiTransformable<TResultModel>>()
401 .Select(x => x.ToApi())
402 .ToList();
403
404 if (resultTransformer != null)
405 foreach (var finalResult in finalResults)
406 await resultTransformer(finalResult);
407
408 var carryTheOne = totalResults % pageSize != 0
409 ? 1
410 : 0;
411 return Json(
412 new PaginatedResponse<TResultModel>
413 {
414 Content = finalResults,
415 PageSize = pageSize,
416 TotalPages = (ushort)(totalResults / pageSize) + carryTheOne,
417 TotalItems = totalResults,
418 });
419 }
