86 {
87 ArgumentNullException.ThrowIfNull(user);
88
89 var now = DateTimeOffset.UtcNow;
90 var nowUnix = now.ToUnixTimeSeconds();
91
92 // this prevents validation conflicts down the line
93 // tldr we can (theoretically) receive a token the same second after we generate it
94 // since unix time rounds down, it looks like it came from before the user changed their password
95 // this happens occasionally in unit tests
96 // just delay a second so we can force a round up
97 var userLastPassworUpdateUnix = user.LastPasswordUpdate?.ToUnixTimeSeconds();
98 DateTimeOffset notBefore;
99 if (nowUnix == userLastPassworUpdateUnix)
100 notBefore = now.AddSeconds(1);
101 else
102 notBefore = now;
103
104 var expiry = now.AddMinutes(oAuth
105 ? securityConfiguration.OAuthTokenExpiryMinutes
106 : securityConfiguration.TokenExpiryMinutes);
107
108 var securityToken = new JwtSecurityToken(
109 tokenHeader,
110 new JwtPayload(
111 ValidationParameters.ValidIssuer,
112 ValidationParameters.ValidAudience,
113 Enumerable.Empty<Claim>(),
114 new Dictionary<string, object>
115 {
116 { JwtRegisteredClaimNames.Sub, user.Id.Value.ToString(CultureInfo.InvariantCulture) },
117 },
118 notBefore.UtcDateTime,
119 expiry.UtcDateTime,
120 now.UtcDateTime));
121
122#pragma warning disable CS0618 // Type or member is obsolete
