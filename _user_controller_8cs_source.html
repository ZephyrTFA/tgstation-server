85 public async ValueTask<IActionResult> Create([FromBody] UserCreateRequest model, CancellationToken cancellationToken)
86 {
87 ArgumentNullException.ThrowIfNull(model);
88
89 if (model.OAuthConnections?.Any(x => x == null) == true)
90 return BadRequest(new ErrorMessageResponse(ErrorCode.ModelValidationFailure));
91
92 if ((model.Password != null && model.SystemIdentifier != null)
93 || (model.Password == null && model.SystemIdentifier == null && model.OAuthConnections?.Any() != true))
94 return BadRequest(new ErrorMessageResponse(ErrorCode.UserMismatchPasswordSid));
95
96 if (model.Group != null && model.PermissionSet != null)
97 return BadRequest(new ErrorMessageResponse(ErrorCode.UserGroupAndPermissionSet));
98
99 model.Name = model.Name?.Trim();
100 if (model.Name?.Length == 0)
101 model.Name = null;
102
103 if (!(model.Name == null ^ model.SystemIdentifier == null))
104 return BadRequest(new ErrorMessageResponse(ErrorCode.UserMismatchNameSid));
105
106 var fail = CheckValidName(model, true);
107 if (fail != null)
108 return fail;
109
110 var totalUsers = await DatabaseContext
111 .Users
112 .AsQueryable()
113 .CountAsync(cancellationToken);
114 if (totalUsers >= generalConfiguration.UserLimit)
115 return Conflict(new ErrorMessageResponse(ErrorCode.UserLimitReached));
116
117 var dbUser = await CreateNewUserFromModel(model, cancellationToken);
118 if (dbUser == null)
119 return this.Gone();
120
121 if (model.SystemIdentifier != null)
122 try
123 {
124 using var sysIdentity = await systemIdentityFactory.CreateSystemIdentity(dbUser, cancellationToken);
125 if (sysIdentity == null)
126 return this.Gone();
127 dbUser.Name = sysIdentity.Username;
128 dbUser.SystemIdentifier = sysIdentity.Uid;
129 }
130 catch (NotImplementedException ex)
131 {
132 return RequiresPosixSystemIdentity(ex);
133 }
134 else if (!(model.Password?.Length == 0 && model.OAuthConnections?.Any() == true))
135 {
136 var result = TrySetPassword(dbUser, model.Password, true);
137 if (result != null)
138 return result;
139 }
140
141 dbUser.CanonicalName = Models.User.CanonicalizeName(dbUser.Name);
142
143 DatabaseContext.Users.Add(dbUser);
144
145 await DatabaseContext.Save(cancellationToken);
146
147 Logger.LogInformation("Created new user {name} ({id})", dbUser.Name, dbUser.Id);
148
149 return Created(dbUser.ToApi());
150 }

171 public async ValueTask<IActionResult> Update([FromBody] UserUpdateRequest model, CancellationToken cancellationToken)
172 {
173 ArgumentNullException.ThrowIfNull(model);
174
175 if (!model.Id.HasValue || model.OAuthConnections?.Any(x => x == null) == true)
176 return BadRequest(new ErrorMessageResponse(ErrorCode.ModelValidationFailure));
177
178 if (model.Group != null && model.PermissionSet != null)
179 return BadRequest(new ErrorMessageResponse(ErrorCode.UserGroupAndPermissionSet));
180
181 var callerAdministrationRights = (AdministrationRights)AuthenticationContext.GetRight(RightsType.Administration);
182 var canEditAllUsers = callerAdministrationRights.HasFlag(AdministrationRights.WriteUsers);
183 var passwordEdit = canEditAllUsers || callerAdministrationRights.HasFlag(AdministrationRights.EditOwnPassword);
184 var oAuthEdit = canEditAllUsers || callerAdministrationRights.HasFlag(AdministrationRights.EditOwnOAuthConnections);
185
186 var originalUser = !canEditAllUsers
187 ? AuthenticationContext.User
188 : await DatabaseContext
189 .Users
190 .AsQueryable()
191 .Where(x => x.Id == model.Id)
192 .Include(x => x.CreatedBy)
193 .Include(x => x.OAuthConnections)
194 .Include(x => x.Group)
195 .ThenInclude(x => x.PermissionSet)
196 .Include(x => x.PermissionSet)
197 .FirstOrDefaultAsync(cancellationToken);
198
199 if (originalUser == default)
200 return NotFound();
201
202 if (originalUser.CanonicalName == Models.User.CanonicalizeName(Models.User.TgsSystemUserName))
203 return Forbid();
204
205 // Ensure they are only trying to edit things they have perms for (system identity change will trigger a bad request)
206 if ((!canEditAllUsers
207 && (model.Id != originalUser.Id
208 || model.Enabled.HasValue
209 || model.Group != null
210 || model.PermissionSet != null
211 || model.Name != null))
212 || (!passwordEdit && model.Password != null)
213 || (!oAuthEdit && model.OAuthConnections != null))
214 return Forbid();
215
216 var originalUserHasSid = originalUser.SystemIdentifier != null;
217 if (originalUserHasSid && originalUser.PasswordHash != null)
218 {
219 // cleanup from https://github.com/tgstation/tgstation-server/issues/1528
220 Logger.LogDebug("System user ID {userId}'s PasswordHash is polluted, updating database.", originalUser.Id);
221 originalUser.PasswordHash = null;
222 originalUser.LastPasswordUpdate = DateTimeOffset.UtcNow;
223 }
224
225 if (model.SystemIdentifier != null && model.SystemIdentifier != originalUser.SystemIdentifier)
226 return BadRequest(new ErrorMessageResponse(ErrorCode.UserSidChange));
227
228 if (model.Password != null)
229 {
230 if (originalUserHasSid)
231 return BadRequest(new ErrorMessageResponse(ErrorCode.UserMismatchPasswordSid));
232
233 var result = TrySetPassword(originalUser, model.Password, false);
234 if (result != null)
235 return result;
236 }
237
238 if (model.Name != null && Models.User.CanonicalizeName(model.Name) != originalUser.CanonicalName)
239 return BadRequest(new ErrorMessageResponse(ErrorCode.UserNameChange));
240
241 if (model.Enabled.HasValue)
242 {
243 if (originalUser.Enabled.Value && !model.Enabled.Value)
244 originalUser.LastPasswordUpdate = DateTimeOffset.UtcNow;
245
246 originalUser.Enabled = model.Enabled.Value;
247 }
248
249 if (model.OAuthConnections != null
250 && (model.OAuthConnections.Count != originalUser.OAuthConnections.Count
251 || !model.OAuthConnections.All(x => originalUser.OAuthConnections.Any(y => y.Provider == x.Provider && y.ExternalUserId == x.ExternalUserId))))
252 {
253 if (originalUser.CanonicalName == Models.User.CanonicalizeName(DefaultCredentials.AdminUserName))
254 return BadRequest(new ErrorMessageResponse(ErrorCode.AdminUserCannotOAuth));
255
256 if (model.OAuthConnections.Count == 0 && originalUser.PasswordHash == null && originalUser.SystemIdentifier == null)
257 return BadRequest(new ErrorMessageResponse(ErrorCode.CannotRemoveLastAuthenticationOption));
258
259 originalUser.OAuthConnections.Clear();
260 foreach (var updatedConnection in model.OAuthConnections)
261 originalUser.OAuthConnections.Add(new Models.OAuthConnection
262 {
263 Provider = updatedConnection.Provider,
264 ExternalUserId = updatedConnection.ExternalUserId,
265 });
266 }
267
268 if (model.Group != null)
269 {
270 originalUser.Group = await DatabaseContext
271 .Groups
272 .AsQueryable()
273 .Where(x => x.Id == model.Group.Id)
274 .Include(x => x.PermissionSet)
275 .FirstOrDefaultAsync(cancellationToken);
276
277 if (originalUser.Group == default)
278 return this.Gone();
279
280 DatabaseContext.Groups.Attach(originalUser.Group);
281 if (originalUser.PermissionSet != null)
282 {
283 Logger.LogInformation("Deleting permission set {0}...", originalUser.PermissionSet.Id);
284 DatabaseContext.PermissionSets.Remove(originalUser.PermissionSet);
285 originalUser.PermissionSet = null;
286 }
287 }
288 else if (model.PermissionSet != null)
289 {
290 if (originalUser.PermissionSet == null)
291 {
292 Logger.LogTrace("Creating new permission set...");
293 originalUser.PermissionSet = new Models.PermissionSet();
294 }
295
296 originalUser.PermissionSet.AdministrationRights = model.PermissionSet.AdministrationRights ?? AdministrationRights.None;
297 originalUser.PermissionSet.InstanceManagerRights = model.PermissionSet.InstanceManagerRights ?? InstanceManagerRights.None;
298
299 originalUser.Group = null;
300 originalUser.GroupId = null;
301 }
302
303 var fail = CheckValidName(model, false);
304 if (fail != null)
305 return fail;
306
307 originalUser.Name = model.Name ?? originalUser.Name;
308
309 await DatabaseContext.Save(cancellationToken);
310
311 Logger.LogInformation("Updated user {0} ({1})", originalUser.Name, originalUser.Id);
312
313 // return id only if not a self update and cannot read users
314 var canReadBack = AuthenticationContext.User.Id == originalUser.Id
315 || callerAdministrationRights.HasFlag(AdministrationRights.ReadUsers);
316 return canReadBack
317 ? Json(originalUser.ToApi())
318 : NoContent();
319 }
