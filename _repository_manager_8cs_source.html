120 public async ValueTask<IRepository> CloneRepository(
121 Uri url,
122 string initialBranch,
123 string username,
124 string password,
125 JobProgressReporter progressReporter,
126 bool recurseSubmodules,
127 CancellationToken cancellationToken)
128 {
129 ArgumentNullException.ThrowIfNull(url);
130 ArgumentNullException.ThrowIfNull(progressReporter);
131
132 logger.LogInformation("Begin clone {url} (Branch: {initialBranch})", url, initialBranch);
133 lock (semaphore)
134 {
135 if (CloneInProgress)
136 throw new JobException(ErrorCode.RepoCloning);
137 CloneInProgress = true;
138 }
139
140 try
141 {
142 using (await SemaphoreSlimContext.Lock(semaphore, cancellationToken))
143 {
144 logger.LogTrace("Semaphore acquired for clone");
145 var repositoryPath = ioManager.ResolvePath();
146 if (!await ioManager.DirectoryExists(repositoryPath, cancellationToken))
147 try
148 {
149 var cloneOptions = new CloneOptions
150 {
151 OnProgress = (a) => !cancellationToken.IsCancellationRequested,
152 OnTransferProgress = (a) =>
153 {
154 var percentage = ((double)a.IndexedObjects + a.ReceivedObjects) / (a.TotalObjects * 2);
155 progressReporter.ReportProgress(percentage);
156 return !cancellationToken.IsCancellationRequested;
157 },
158 RecurseSubmodules = recurseSubmodules,
159 OnUpdateTips = (a, b, c) => !cancellationToken.IsCancellationRequested,
160 RepositoryOperationStarting = (a) => !cancellationToken.IsCancellationRequested,
161 BranchName = initialBranch,
162 CredentialsProvider = repositoryFactory.GenerateCredentialsHandler(username, password),
163 };
164
165 await repositoryFactory.Clone(
166 url,
167 cloneOptions,
168 repositoryPath,
169 cancellationToken);
170 }
171 catch
172 {
173 try
174 {
175 logger.LogTrace("Deleting partially cloned repository...");
176
177 // DCT: Cancellation token is for job, operation must run regardless
178 await ioManager.DeleteDirectory(repositoryPath, CancellationToken.None);
179 }
180 catch (Exception innerException)
181 {
182 logger.LogError(innerException, "Error deleting partially cloned repository!");
183 }
184
185 throw;
186 }
187 else
188 {
189 logger.LogDebug("Repository exists, clone aborted!");
190 return null;
191 }
192 }
193 }
194 finally
195 {
196 logger.LogTrace("Semaphore released after clone");
197 CloneInProgress = false;
198 }
199
200 logger.LogInformation("Clone complete!");
201 return await LoadRepository(cancellationToken);
202 }

205 public async ValueTask<IRepository> LoadRepository(CancellationToken cancellationToken)
206 {
207 logger.LogTrace("Begin LoadRepository...");
208 lock (semaphore)
209 if (CloneInProgress)
210 throw new JobException(ErrorCode.RepoCloning);
211
212 try
213 {
214 await semaphore.WaitAsync(cancellationToken);
215 try
216 {
217 logger.LogTrace("Semaphore acquired for load");
218 var libGit2Repo = await repositoryFactory.CreateFromPath(ioManager.ResolvePath(), cancellationToken);
219
220 return new Repository(
221 libGit2Repo,
222 commands,
223 ioManager,
224 eventConsumer,
225 repositoryFactory,
226 postWriteHandler,
227 gitRemoteFeaturesFactory,
228 repositoryLogger,
229 generalConfiguration,
230 () =>
231 {
232 logger.LogTrace("Releasing semaphore due to Repository disposal...");
233 semaphore.Release();
234 });
235 }
236 catch
237 {
238 logger.LogTrace("Releasing semaphore as load failed");
239 semaphore.Release();
240 throw;
241 }
242 }
243 catch (RepositoryNotFoundException e)
244 {
245 logger.LogTrace(e, "Repository not found!");
246 return null;
247 }
248 }
