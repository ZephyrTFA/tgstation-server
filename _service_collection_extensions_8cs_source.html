145 public static IServiceCollection UseStandardConfig<TConfig>(this IServiceCollection serviceCollection, IConfiguration configuration)
146 where TConfig : class
147 {
148 ArgumentNullException.ThrowIfNull(serviceCollection);
149 ArgumentNullException.ThrowIfNull(configuration);
150
151 const string SectionFieldName = nameof(GeneralConfiguration.Section);
152
153 var configType = typeof(TConfig);
154 var sectionField = configType.GetField(SectionFieldName) ?? throw new InvalidOperationException(
155 String.Format(CultureInfo.InvariantCulture, "{0} has no {1} field!", configType, SectionFieldName));
156 var stringType = typeof(string);
157 if (sectionField.FieldType != stringType)
158 throw new InvalidOperationException(String.Format(CultureInfo.InvariantCulture, "{0} has invalid {1} field type, must be {2}!", configType, SectionFieldName, stringType));
159
160 var sectionName = (string)sectionField.GetValue(null);
161
162 return serviceCollection.Configure<TConfig>(configuration.GetSection(sectionName));
163 }

175 public static IServiceCollection SetupLogging(
176 this IServiceCollection serviceCollection,
177 Action<LoggerConfiguration> configurationAction,
178 Action<LoggerSinkConfiguration> sinkConfigurationAction = null,
179 ElasticsearchSinkOptions elasticsearchSinkOptions = null,
180 InternalConfiguration internalConfiguration = null,
181 FileLoggingConfiguration fileLoggingConfiguration = null)
182 {
183 if (internalConfiguration != null)
184 ArgumentNullException.ThrowIfNull(fileLoggingConfiguration);
185
186 return serviceCollection.AddLogging(builder =>
187 {
188 builder.ClearProviders();
189
190 var configuration = new LoggerConfiguration()
191 .MinimumLevel
192 .Verbose();
193
194 configurationAction?.Invoke(configuration);
195
196 configuration
197 .Enrich.FromLogContext()
198 .WriteTo
199 .Async(sinkConfiguration =>
200 {
201 var template = "[{Timestamp:HH:mm:ss}] {Level:w3}: {SourceContext:l} ("
202 + SerilogContextHelper.Template
203 + "){NewLine} {Message:lj}{NewLine}{Exception}";
204
205 if (!((internalConfiguration?.UsingSystemD ?? false) && !fileLoggingConfiguration.Disable))
206 sinkConfiguration.Console(outputTemplate: template, formatProvider: CultureInfo.InvariantCulture);
207 sinkConfigurationAction?.Invoke(sinkConfiguration);
208 });
209
210 if (elasticsearchSinkOptions != null)
211 configuration.WriteTo.Elasticsearch(elasticsearchSinkOptions);
212
213 builder.AddSerilog(configuration.CreateLogger(), true);
214
215 if (Debugger.IsAttached)
216 builder.AddDebug();
217
218 if (additionalLoggerProvider != null)
219 builder.Services.TryAddEnumerable(additionalLoggerProvider);
220 });
