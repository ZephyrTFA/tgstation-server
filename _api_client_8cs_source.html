310 HttpContent? content,
311 HttpMethod method,
312 long? instanceId,
313 bool tokenRefresh,
314 CancellationToken cancellationToken)
315 {
316 if (route == null)
317 throw new ArgumentNullException(nameof(route));
318 if (method == null)
319 throw new ArgumentNullException(nameof(method));
320 if (content == null && (method == HttpMethod.Post || method == HttpMethod.Put))
321 throw new InvalidOperationException("content cannot be null for POST or PUT!");
322
323 HttpResponseMessage response;
324 var fullUri = new Uri(Url, route);
325 var serializerSettings = GetSerializerSettings();
326 var fileDownload = typeof(TResult) == typeof(Stream);
327 using (var request = new HttpRequestMessage(method, fullUri))
328 {
329 if (content != null)
330 request.Content = content;
331
332 try
333 {
334 var headersToUse = tokenRefresh ? tokenRefreshHeaders! : headers;
335 headersToUse.SetRequestHeaders(request.Headers, instanceId);
336
337 if (authless)
338 request.Headers.Remove(HeaderNames.Authorization);
339
340 if (fileDownload)
341 request.Headers.Accept.Add(new MediaTypeWithQualityHeaderValue(MediaTypeNames.Application.Octet));
342
343 await ValueTaskExtensions.WhenAll(requestLoggers.Select(x => x.LogRequest(request, cancellationToken))).ConfigureAwait(false);
344
345 response = await httpClient.SendAsync(request, HttpCompletionOption.ResponseHeadersRead, cancellationToken).ConfigureAwait(false);
346 }
347 finally
348 {
349 // prevent content param from getting disposed
350 request.Content = null;
351 }
352 }
353
354 try
355 {
356 await ValueTaskExtensions.WhenAll(requestLoggers.Select(x => x.LogResponse(response, cancellationToken))).ConfigureAwait(false);
357
358 // just stream
359 if (fileDownload && response.IsSuccessStatusCode)

361 }
362 catch

401 async ValueTask<bool> RefreshToken(CancellationToken cancellationToken)
402 {
403 if (tokenRefreshHeaders == null)
404 return false;
405
406 var startingToken = headers.Token;
407 await semaphoreSlim.WaitAsync(cancellationToken).ConfigureAwait(false);
408 try
409 {
410 if (startingToken != headers.Token)
411 return true;
412
413 var token = await RunRequest<object, TokenResponse>(Routes.Root, new object(), HttpMethod.Post, null, true, cancellationToken).ConfigureAwait(false);
414 headers = new ApiHeaders(headers.UserAgent!, token.Bearer!);
415 }
416 catch (ClientException)
417 {
418 return false;
419 }
420 finally
421 {
422 semaphoreSlim.Release();
423 }
424
425 return true;
