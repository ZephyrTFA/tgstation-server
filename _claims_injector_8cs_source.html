73
74 var enumerator = Enum.GetValues(typeof(RightsType));
75 var claims = new List<Claim>();
76 foreach (RightsType rightType in enumerator)
77 {
78 // if there's no instance user, do a weird thing and add all the instance roles
79 // we need it so we can get to OnActionExecutionAsync where we can properly decide between BadRequest and Forbid
80 // if user is null that means they got the token with an expired password
81 var rightAsULong = authenticationContext.User == null

90 }
