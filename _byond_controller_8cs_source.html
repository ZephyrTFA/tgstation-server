139#pragma warning restore CA1506

143 var uploadingZip = model.UploadCustomZip == true;
144
145 if (model.Version == null
146 || model.Version.Revision != -1
147 || (uploadingZip && model.Version.Build > 0))
148 return BadRequest(new ErrorMessageResponse(ErrorCode.ModelValidationFailure));
149

151
152 var userByondRights = AuthenticationContext.InstancePermissionSet.ByondRights.Value;
153 if ((!userByondRights.HasFlag(ByondRights.InstallOfficialOrChangeActiveVersion) && !uploadingZip)
154 || (!userByondRights.HasFlag(ByondRights.InstallCustomVersion) && uploadingZip))
155 return Forbid();
156
157 // remove cruff fields
158 var result = new ByondInstallResponse();
159 return await WithComponentInstance(
160 async instance =>
161 {
162 var byondManager = instance.ByondManager;
163 var versionAlreadyInstalled = !uploadingZip && byondManager.InstalledVersions.Any(x => x == version);
164 if (versionAlreadyInstalled)
165 {
166 Logger.LogInformation(
167 "User ID {userId} changing instance ID {instanceId} BYOND version to {newByondVersion}",
168 AuthenticationContext.User.Id,
169 Instance.Id,
170 version);
171
172 try
173 {
174 await byondManager.ChangeVersion(null, version, null, false, cancellationToken);
175 }
176 catch (InvalidOperationException ex)
177 {
178 Logger.LogDebug(
179 ex,
180 "Race condition: BYOND version {version} uninstalled before we could switch to it. Creating install job instead...",
181 version);
182 versionAlreadyInstalled = false;
183 }
184 }
185
186 if (!versionAlreadyInstalled)
187 {
188 if (version.Build > 0)
189 return BadRequest(new ErrorMessageResponse(ErrorCode.ByondNonExistentCustomVersion));
190
191 Logger.LogInformation(
192 "User ID {userId} installing BYOND version to {newByondVersion} on instance ID {instanceId}",
193 AuthenticationContext.User.Id,
194 version,
195 Instance.Id);
196
197 // run the install through the job manager
198 var job = Job.Create(
199 uploadingZip
200 ? JobCode.ByondCustomInstall
201 : JobCode.ByondOfficialInstall,
202 AuthenticationContext.User,
203 Instance,
204 ByondRights.CancelInstall);

208 if (uploadingZip)
209 fileUploadTicket = fileTransferService.CreateUpload(FileUploadStreamKind.None);
210
211 try

214 job,

240 jobCancellationToken);
