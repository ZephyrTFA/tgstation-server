196 var pid = handle.Id;

198
199 // once we obtain these handles we're responsible for them

201 using var stdErrHandle = handle.StandardError;
202 Task<string> outputReadTask = null, errorReadTask = null;
203 bool outputOpen = true, errorOpen = true;
204 async Task<string> GetNextLine()
205 {
206#if NET7_0_OR_GREATER
207#error ReadLineAsync supports cancellation now
208#endif
209 if (outputOpen && outputReadTask == null)

211
212 if (errorOpen && errorReadTask == null)
213 errorReadTask = stdErrHandle.ReadLineAsync();
214
215 var completedTask = await Task.WhenAny(outputReadTask ?? errorReadTask, errorReadTask ?? outputReadTask).WaitAsync(disposeToken);
216 var line = await completedTask;
217 if (completedTask == outputReadTask)
218 {
219 outputReadTask = null;
220 if (line == null)
221 outputOpen = false;
222 }
223 else
224 {
225 errorReadTask = null;
226 if (line == null)
227 errorOpen = false;
228 }
229
230 if (line == null && (errorOpen || outputOpen))
231 return await GetNextLine();
232
233 return line;
234 }
235
236 await using var fileStream = fileRedirect != null ? ioManager.CreateAsyncSequentialWriteStream(fileRedirect) : null;
237 await using var writer = fileStream != null ? new StreamWriter(fileStream) : null;
238
239 string text;
